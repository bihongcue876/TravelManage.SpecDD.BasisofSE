class BaseTest(unittest.TestCase):
    @classmethod
    def setUpClass(cls):
        Base.metadata.create_all(bind=_engine)

    def setUp(self):
        Base.metadata.drop_all(bind=_engine)
        Base.metadata.create_all(bind=_engine)
        self.db = _TestingSession()
        def override_get_db():
            try:
                yield self.db
            finally:
                pass
        app.dependency_overrides[get_db] = override_get_db
        self.client = TestClient(app, base_url="http://test")
        self.route = self._seed_route()

    def tearDown(self):
        app.dependency_overrides.clear()
        self.db.close()
        Base.metadata.drop_all(bind=_engine)
```

// 辅助组件：自动触发登录
function LoginAction({ username, password }: { username: string; password: string }) {
  const { login } = useAuth()
  useEffect(() => {
    login(username, password).catch(() => {})
  }, [])
  return null
}

it('登录成功 → 设置 token、更新 user、isAuthenticated=true', async () => {
  // /api/auth/login 返回 token
  mockFetchOnce({ access_token: 'test-token-123', username: 'admin', name: '管理员', role: 'admin' })
  // login() 内部 fetchUser(/api/auth/me)
  mockFetchOnce({ id: 1, username: 'admin', name: '管理员', role: 'admin', email: 'admin@test.com', phone: null })
  // setToken 触发 useEffect 再调一次 fetchUser
  mockFetchOnce({ id: 1, username: 'admin', name: '管理员', role: 'admin', email: null, phone: null })

  renderWithAuth(<><TestConsumer /><LoginAction username="admin" password="123456" /></>)
  await waitFor(() => {
    expect(screen.getByTestId('authenticated')).toHaveTextContent('true')
  })
  expect(localStorage.getItem('access_token')).toBe('test-token-123')
})
```

**mock 策略**：使用 `vi.fn()` 替代全局 `fetch`，通过 `mockResolvedValueOnce` 按调用顺序返回不同响应，涵盖 login 双阶段（/login 获取 token → /me 获取用户信息）以及 useEffect 触发的重复请求。

it('localStorage 有 token → Authorization header 正确', async () => {
  await import('./api')
  const { fulfilled } = mockRequestInterceptor.handlers[0]
  localStorageMock.setItem('access_token', 'my-token')
  const config = { headers: {} }
  const result = fulfilled(config)

it('非登录接口 401 → 清除 token、触发 auth:logout 事件', async () => {
  await import('./api')
  const { rejected } = mockResponseInterceptor.handlers[0]
